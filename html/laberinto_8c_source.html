85void laberinto_set_muro(uint8_t fila, uint8_t columna, brujula direccion)

120void laberinto_recalcular_pesos(void)
121{
122 bool cambio_detectado = true;
123 uint8_t iteraciones = 0;
124 const uint8_t MAX_ITERACIONES = 20; // Evitar bucles infinitos Protección contra boludos
125
126 // Algoritmo Flood Fill iterativo
127 while (cambio_detectado && iteraciones < MAX_ITERACIONES)
128 {
129 cambio_detectado = false;
130 iteraciones++;
131
132 for (uint8_t fila = 1; fila <= TAMAÑO_LABERINTO; fila++) // se propaga desde (1,1) hacia afuera
133 {
134 for (uint8_t columna = 1; columna <= TAMAÑO_LABERINTO; columna++)
135 {
136
137 // No recalcular la meta
138 if (fila == POSICION_META_FILA && columna == POSICION_META_COLUMNA)
139 {
140 continue;
141 }
142
143 uint8_t peso_minimo = PESO_MAXIMO;
144
145 // Verificar todas las direcciones adyacentes
146 for (brujula dir = norte; dir <= oeste; dir++)
147 {
148
149 // Saltar si hay muro en esta dirección
150 if (laberinto[fila - 1][columna - 1].muros[dir])
151 {
152 continue;
153 }
154
155 posicion_t pos_adyacente = laberinto_get_posicion_adyacente(
156 (posicion_t){fila, columna}, dir);
157
158 if (laberinto_posicion_valida(pos_adyacente.fila, pos_adyacente.columna))
159 {
160 uint8_t peso_adyacente = laberinto_get_peso(
161 pos_adyacente.fila, pos_adyacente.columna);
162
163 if (peso_adyacente < peso_minimo)
164 {
165 peso_minimo = peso_adyacente;
166 }
167 }
168 }
169
170 // Nuevo peso = menor peso adyacente + 1
171 uint8_t nuevo_peso = (peso_minimo == PESO_MAXIMO) ? PESO_MAXIMO : peso_minimo + 1;
172
173 // Actualizar si hay cambio
174 if (nuevo_peso != laberinto[fila - 1][columna - 1].peso)
175 {
176 laberinto[fila - 1][columna - 1].peso = nuevo_peso;
177 cambio_detectado = true;
178 }
179 }
180 }
181 }
182}
