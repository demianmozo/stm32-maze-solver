48void set_motor_izq(motor_estado_t estado, uint16_t pwm)
49{
50 switch (estado)

57 case MOTOR_RETROCESO:

62 case MOTOR_FRENADO:

66 pwm = 0; // Forzar PWM a 0 en frenado
67 break;
68 }

71 __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_3, pwm);
